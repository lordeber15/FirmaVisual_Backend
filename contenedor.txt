services:
  db:
    image: postgres:16-alpine
    container_name: rpd-postgres
    restart: unless-stopped
    environment:
      POSTGRES_USER: rpd_user
      POSTGRES_PASSWORD: rpd_pass
      POSTGRES_DB: rpd_db
    ports:
      - "5435:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U rpd_user -d rpd_db"]
      interval: 10s
      timeout: 5s
      retries: 5
      start_period: 10s

  backend:
    build:
      context: ./backend
      target: development
    container_name: rpd-backend
    restart: unless-stopped
    volumes:
      - ./backend:/app
      - /app/node_modules
      - uploads_data:/app/uploads
    ports:
      - "3001:3001"
    environment:
      - NODE_ENV=development
      - PORT=3001
      - DB_HOST=db
      - DB_USER=rpd_user
      - DB_PASSWORD=rpd_pass
      - DB_NAME=rpd_db
      - JWT_SECRET=rpd-dev-secret-change-in-production-2024
      - CORS_ORIGIN=http://localhost:3005,http://192.168.15.12:3005
    depends_on:
      db:
        condition: service_healthy
    healthcheck:
      test: ["CMD", "wget", "--no-verbose", "--tries=1", "--spider", "http://localhost:3001/api/health"]
      interval: 15s
      timeout: 5s
      retries: 3
      start_period: 20s

  frontend:
    build:
      context: ./frontend
      target: development
    container_name: rpd-frontend
    restart: unless-stopped
    volumes:
      - ./frontend:/app
      - /app/node_modules
    ports:
      - "3005:3000"
    depends_on:
      backend:
        condition: service_healthy

volumes:
  postgres_data:
  uploads_data:
